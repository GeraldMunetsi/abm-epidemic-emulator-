def cohen_d(g1, g2):
    n1, n2 = len(g1), len(g2)
    pooled_sd = np.sqrt(((n1-1)*g1.var(ddof=1) + (n2-1)*g2.var(ddof=1)) / (n1+n2-2))
    return (g1.mean() - g2.mean()) / pooled_sd
